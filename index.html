    <script>

        let currentQuestion = 1;

        let answers = {};


        function updateStudentName() {

            const studentName = document.getElementById('studentName').value || 'Sam';

            const placeholders = document.querySelectorAll('.student-placeholder');

            placeholders.forEach(placeholder => {

                placeholder.textContent = studentName;

            });

        }


        // Update student name as user types

        document.getElementById('studentName').addEventListener('input', updateStudentName);


        function hideAllQuestions() {

            for (let i = 1; i <= 8; i++) {

                document.getElementById(`question${i}`).classList.add('hidden');

                document.getElementById(`question${i}`).classList.remove('active');

            }

            document.getElementById('question5b').classList.add('hidden');

            document.getElementById('question5b').classList.remove('active');

            document.getElementById('phase2').classList.add('hidden');

            document.getElementById('criteriaHeader').classList.add('hidden');

        }


        function showQuestion(questionNum) {

            hideAllQuestions();

            const questionId = `question${questionNum}`;

            document.getElementById(questionId).classList.remove('hidden');

            document.getElementById(questionId).classList.add('active');

            currentQuestion = questionNum;

        }


        function showResult(message, type) {

            hideAllQuestions();

            const result = document.getElementById('result');

            result.innerHTML = message;

            result.className = `result ${type}`;

            result.classList.remove('hidden');

            document.getElementById('restartBtn').classList.remove('hidden');

        }


        function handleAnswer(questionNum, answer) {

            answers[questionNum] = answer;

            

            // Update button styles

            const buttons = document.querySelectorAll(`#question${questionNum} .answer-btn`);

            buttons.forEach(btn => btn.classList.remove('selected'));

            event.target.classList.add('selected');


            setTimeout(() => {

                processAnswer(questionNum, answer);

            }, 500);

        }


        function processAnswer(questionNum, answer) {

            switch(questionNum) {

                case 1:

                    if (answer === 'A') {

                        showResult('🎉 Great! Since this is the first time, no "Change of School" paperwork is needed. You can proceed with the standard registration forms.', 'success');

                    } else {

                        showQuestion(2);

                    }

                    break;

                

                case 2:

                    if (answer === 'A') {

                        showResult('🎉 Great! Since there\'s no change of school, no "Change of School" paperwork is needed. You can proceed with the standard registration forms.', 'success');

                    } else {

                        document.getElementById('phase1').classList.add('hidden');

                        document.getElementById('phase2').classList.remove('hidden');

                        showQuestion(3);

                    }

                    break;

                

                case 3:

                    if (answer === 'A') {

                        document.getElementById('criteriaHeader').classList.remove('hidden');

                        showQuestion(6);

                    } else {

                        showQuestion(4);

                    }

                    break;

                

                case 4:

                    if (answer === 'A') {

                        document.getElementById('criteriaHeader').classList.remove('hidden');

                        showQuestion(6);

                    } else {

                        showQuestion(5);

                    }

                    break;

                

                case 5:

                    if (answer === 'A') {

                        showResult('🚨 <strong>Criteria 2 Track</strong>', 'warning');

                    } else {

                        showQuestion('5b');

                    }

                    break;

                

                case '5b':

                    if (answer === 'A') {

                        showResult('🚨 <strong>Criteria 2 Track</strong>', 'warning');

                    } else {

                        document.getElementById('criteriaHeader').classList.remove('hidden');

                        showQuestion(6);

                    }

                    break;

                

                case 6:

                    if (answer === 'A') {

                        showResult('🚨 <strong>Criteria 2 Track</strong>', 'warning');

                    } else {

                        // Check if this is coming from the bonafide move path (questions 3,4,5,5b all "Yes")

                        if (answers[3] === 'B' && answers[4] === 'B' && answers[5] === 'B' && answers['5b'] === 'B') {

                            showResult('✅ <strong>Bonafide Move Confirmed!</strong>', 'success');

                        } else {

                            showQuestion(7);

                        }

                    }

                    break;

                

                case 7:

                    if (answer === 'A') {

                        showResult('🚨 <strong>Criteria 2 Track</strong>', 'warning');

                    } else {

                        showQuestion(8);

                    }

                    break;

                

                case 8:

                    if (answer === 'A') {

                        showResult('🚨 <strong>Criteria 2 Track</strong>', 'warning');

                    } else {

                        showResult('✅ <strong>Criteria 1 Track</strong>', 'success');

                    }

                    break;

            }

        }


        function restart() {

            answers = {};

            currentQuestion = 1;

            document.getElementById('studentName').value = 'Sam';

            updateStudentName();

            

            // Hide all elements

            hideAllQuestions();

            document.getElementById('result').classList.add('hidden');

            document.getElementById('restartBtn').classList.add('hidden');

            document.getElementById('phase1').classList.remove('hidden');

            document.getElementById('phase2').classList.add('hidden');

            document.getElementById('criteriaHeader').classList.add('hidden');

            

            // Show first question

            document.getElementById('question1').classList.remove('hidden');

            document.getElementById('question1').classList.add('active');

            

            // Reset all button selections

            document.querySelectorAll('.answer-btn').forEach(btn => {

                btn.classList.remove('selected');

            });

        }
